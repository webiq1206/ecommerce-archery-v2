# Stage 1 — Replit Build Instructions

## Premium Archery E-Commerce Platform: Architecture, Scaffolding & Foundation

-----

## Project Overview

You are scaffolding a production-grade, premium e-commerce platform for an archery and hunting gear company. This is a Next.js 14+ application using the App Router, server-side rendering throughout, a PostgreSQL database via Prisma ORM, and a fully custom admin panel. The stack must be clean, scalable, and ready for a complete frontend and business logic build-out in Stage 2.

Do not build placeholder UI. Build real structure: routes, schemas, API handlers, middleware, auth, and component skeletons that Cursor can build on top of. Every file you create should be real, functional, and well-organized.

-----

## Technology Stack

All services are either completely free, open-source, or free-tier-first with no monthly minimum. The only cost-at-use service is Stripe (transaction fees only, no monthly fee) and the Anthropic API (pay per token, no subscription required).

- **Framework**: Next.js 14+ with App Router (never Pages Router, never Vite)
- **Language**: TypeScript throughout — no JavaScript files
- **Database**: PostgreSQL via Prisma ORM — free to self-host or use Supabase free tier (500MB, no credit card required)
- **Auth**: NextAuth.js v5 — free, open-source
- **Payments**: Stripe — no monthly fee, transaction fees only (2.9% + 30¢). No viable free alternative for real card processing.
- **File Storage**: UploadThing — free tier (2GB storage, unlimited uploads within bandwidth). Native Next.js integration, no credit card to start.
- **Email**: Resend — free tier (3,000 emails/month, 100/day). More than enough to launch and validate the business.
- **Search**: PostgreSQL Full-Text Search (built-in, zero cost) — uses `tsvector`/`tsquery` directly in Prisma raw queries. Handles product search, filters, and instant search well at this scale. Zero additional infrastructure.
- **AI**: Anthropic Claude API — pay per token, no monthly fee. Only AI cost in the stack.
- **Styling**: Tailwind CSS v3 — free, open-source
- **State Management**: Zustand — free, open-source
- **Forms**: React Hook Form + Zod — free, open-source
- **Analytics**: PostHog — free tier (1 million events/month, session recordings, funnels). No credit card required.
- **Rich Text Editor**: TipTap — free open-source core
- **SMS**: Scaffold only — no service integrated until needed
- **Cache / Rate Limiting**: In-memory cache via `lru-cache` (no external service needed at this scale). Cart persistence handled by Zustand + localStorage. Session handled by NextAuth.
- **Deployment Target**: Vercel — free Hobby tier works for development; Pro ($20/month) needed for production cron jobs and higher limits. Alternatively: Railway (free tier) or Render (free tier).

### `/app/api/products/route.ts`

```typescript
import { NextRequest, NextResponse } from 'next/server'
import { prisma } from '@/lib/db/prisma'
import { getServerSession } from 'next-auth'
import { authConfig } from '@/lib/auth/config'
import { z } from 'zod'

const ProductQuerySchema = z.object({
  page:       z.coerce.number().min(1).default(1),
  limit:      z.coerce.number().min(1).max(100).default(24),
  category:   z.string().optional(),
  brand:      z.string().optional(),
  collection: z.string().optional(),
  status:     z.enum(['DRAFT', 'ACTIVE', 'ARCHIVED']).optional(),
  search:     z.string().optional(),
  sort:       z.enum(['newest', 'price_asc', 'price_desc', 'name_asc']).default('newest'),
  featured:   z.coerce.boolean().optional(),
})

export async function GET(req: NextRequest) {
  try {
    const params = Object.fromEntries(req.nextUrl.searchParams)
    const query = ProductQuerySchema.parse(params)
    const skip = (query.page - 1) * query.limit

    const where: any = {}
    if (query.status) where.status = query.status
    else where.status = 'ACTIVE'
    if (query.brand) where.brand = { slug: query.brand }
    if (query.featured !== undefined) where.isFeatured = query.featured
    if (query.search) {
      where.OR = [
        { name: { contains: query.search, mode: 'insensitive' } },
        { sku: { contains: query.search, mode: 'insensitive' } },
      ]
    }
    if (query.category) {
      where.categories = { some: { category: { slug: query.category } } }
    }
    if (query.collection) {
      where.collections = { some: { collection: { slug: query.collection } } }
    }

    const orderBy: any = {
      newest: { createdAt: 'desc' },
      price_asc: { price: 'asc' },
      price_desc: { price: 'desc' },
      name_asc: { name: 'asc' },
    }[query.sort]

    const [products, total] = await Promise.all([
      prisma.product.findMany({
        where,
        skip,
        take: query.limit,
        orderBy,
        include: {
          images: { orderBy: { sortOrder: 'asc' }, take: 2 },
          brand: { select: { name: true, slug: true } },
          categories: {
            where: { isPrimary: true },
            include: { category: { select: { name: true, slug: true } } },
          },
          _count: { select: { reviews: { where: { isApproved: true } } } },
        },
      }),
      prisma.product.count({ where }),
    ])

    return NextResponse.json({
      products,
      total,
      page: query.page,
      limit: query.limit,
      totalPages: Math.ceil(total / query.limit),
    })
  } catch (err) {
    console.error('[GET /api/products]', err)
    return NextResponse.json({ error: 'Failed to fetch products' }, { status: 500 })
  }
}

export async function POST(req: NextRequest) {
  try {
    const session = await getServerSession(authConfig)
    if (!session || !['ADMIN', 'SUPER_ADMIN'].includes(session.user.role)) {
      return NextResponse.json({ error: 'Unauthorized' }, { status: 401 })
    }
    const body = await req.json()
    // Full product creation logic built in Stage 2
    const product = await prisma.product.create({ data: body })
    return NextResponse.json(product, { status: 201 })
  } catch (err) {
    return NextResponse.json({ error: 'Failed to create product' }, { status: 500 })
  }
}
```

### `/app/api/fulfillment/route.ts`

Build the fulfillment routing engine. This is a critical business feature:

```typescript
import { NextRequest, NextResponse } from 'next/server'
import { prisma } from '@/lib/db/prisma'
import { sendFulfillmentEmail } from '@/lib/fulfillment/email'
import { getServerSession } from 'next-auth'
import { authConfig } from '@/lib/auth/config'

/**
 * POST /api/fulfillment
 * Called after successful payment. Accepts orderId.
 * 1. Load order with all items and their distributor assignments
 * 2. Group items by distributor
 * 3. Send one fulfillment email per distributor with only their items
 * 4. Log all fulfillment events
 */
export async function POST(req: NextRequest) {
  try {
    const { orderId } = await req.json()

    const order = await prisma.order.findUnique({
      where: { id: orderId },
      include: {
        items: {
          include: {
            product: {
              include: { distributor: true },
            },
            variant: true,
          },
        },
        fulfillmentLogs: true,
      },
    })

    if (!order) {
      return NextResponse.json({ error: 'Order not found' }, { status: 404 })
    }

    // Group items by distributor
    const distributorMap = new Map<string, {
      distributor: any,
      items: any[]
    }>()

    const unassignedItems: any[] = []

    for (const item of order.items) {
      const distributor = item.product.distributor
      if (!distributor) {
        unassignedItems.push(item)
        continue
      }
      if (!distributorMap.has(distributor.id)) {
        distributorMap.set(distributor.id, { distributor, items: [] })
      }
      distributorMap.get(distributor.id)!.items.push(item)
    }

    const logs = []

    // Send fulfillment email to each distributor
    for (const [distributorId, { distributor, items }] of distributorMap) {
      try {
        await sendFulfillmentEmail({
          order,
          distributor,
          items,
        })

        const log = await prisma.fulfillmentLog.create({
          data: {
            orderId: order.id,
            distributorId,
            status: 'EMAIL_SENT',
            emailSentAt: new Date(),
            emailRecipients: [distributor.email, ...distributor.ccEmails],
            itemsSent: items.map(i => ({
              productId: i.productId,
              variantId: i.variantId,
              sku: i.sku,
              name: i.name,
              qty: i.quantity,
              options: i.options,
            })),
          },
        })
        logs.push(log)
      } catch (emailError) {
        await prisma.fulfillmentLog.create({
          data: {
            orderId: order.id,
            distributorId,
            status: 'FAILED',
            itemsSent: items.map(i => ({ sku: i.sku, qty: i.quantity })),
            notes: String(emailError),
          },
        })
      }
    }

    // If unassigned items exist, alert the store admin
    if (unassignedItems.length > 0) {
      console.warn(`[FULFILLMENT] Order ${order.orderNumber} has ${unassignedItems.length} items with no distributor assigned.`)
      // TODO: Send internal alert email in Stage 2
    }

    return NextResponse.json({ success: true, logsCreated: logs.length, unassignedCount: unassignedItems.length })
  } catch (err) {
    console.error('[FULFILLMENT ERROR]', err)
    return NextResponse.json({ error: 'Fulfillment routing failed' }, { status: 500 })
  }
}
```

```typescript
export interface CartItem {
  id: string
  productId: string
  variantId?: string
  slug: string
  name: string
  sku: string
  price: number
  quantity: number
  imageUrl?: string
  options?: Record<string, string>
}

interface CartStore {
  items: CartItem[]
  isOpen: boolean
  addItem: (item: CartItem) => void
  removeItem: (id: string) => void
  updateQuantity: (id: string, quantity: number) => void
  clearCart: () => void
  openCart: () => void
  closeCart: () => void
  get itemCount(): number
  get subtotal(): number
}

export const useCartStore = create<CartStore>()(
  persist(
    (set, get) => ({
      items: [],
      isOpen: false,

      addItem: (item) => set((state) => {
        const existing = state.items.find(
          (i) => i.productId === item.productId && i.variantId === item.variantId
        )
        if (existing) {
          return {
            items: state.items.map((i) =>
              i.id === existing.id
                ? { ...i, quantity: i.quantity + item.quantity }
                : i
            ),
            isOpen: true,
          }
        }
        return { items: [...state.items, item], isOpen: true }
      }),

      removeItem: (id) => set((state) => ({
        items: state.items.filter((i) => i.id !== id),
      })),

      updateQuantity: (id, quantity) => set((state) => ({
        items: quantity <= 0
          ? state.items.filter((i) => i.id !== id)
          : state.items.map((i) => i.id === id ? { ...i, quantity } : i),
      })),

      clearCart: () => set({ items: [] }),
      openCart: () => set({ isOpen: true }),
      closeCart: () => set({ isOpen: false }),

      get itemCount() {
        return get().items.reduce((sum, i) => sum + i.quantity, 0)
      },

      get subtotal() {
        return get().items.reduce((sum, i) => sum + i.price * i.quantity, 0)
      },
    }),
    {
      name: 'archery-cart',
      storage: createJSONStorage(() => localStorage),
    }
  )
```

### PostgreSQL Full-Text Search — `/app/api/search/route.ts`

This replaces Algolia entirely. PostgreSQL’s built-in `tsvector`/`tsquery` handles product search with zero external dependencies or cost. Performance is excellent for catalogs under 100k products.

```typescript
import { NextRequest, NextResponse } from 'next/server'
import { prisma } from '@/lib/db/prisma'

export async function GET(req: NextRequest) {
  const query = req.nextUrl.searchParams.get('q')?.trim()
  const limit = Math.min(Number(req.nextUrl.searchParams.get('limit') || '12'), 48)
  const page = Number(req.nextUrl.searchParams.get('page') || '1')
  const skip = (page - 1) * limit

  if (!query || query.length < 2) {
    return NextResponse.json({ results: [], total: 0 })
  }

  try {
    // Use Prisma raw query for full-text search with ranking
    // ts_rank orders results by relevance — exact matches rank higher than partial
    const results = await prisma.$queryRaw<any[]>`
      SELECT
        p.id,
        p.name,
        p.slug,
        p.price,
        p."compareAtPrice",
        p."shortDescription",
        b.name as "brandName",
        b.slug as "brandSlug",
        ts_rank(
          to_tsvector('english', p.name || ' ' || COALESCE(p."shortDescription", '') || ' ' || COALESCE(b.name, '')),
          plainto_tsquery('english', ${query})
        ) as rank
      FROM products p
      LEFT JOIN brands b ON p."brandId" = b.id
      WHERE
        p.status = 'ACTIVE'
        AND to_tsvector('english', p.name || ' ' || COALESCE(p."shortDescription", '') || ' ' || COALESCE(b.name, ''))
            @@ plainto_tsquery('english', ${query})
      ORDER BY rank DESC
      LIMIT ${limit}
      OFFSET ${skip}
    `

    // Get primary image for each result
    const productIds = results.map(r => r.id)
    const images = await prisma.productImage.findMany({
      where: { productId: { in: productIds }, sortOrder: 0 },
      select: { productId: true, url: true, altText: true },
    })
    const imageMap = new Map(images.map(i => [i.productId, i]))

    // Count total matches for pagination
    const countResult = await prisma.$queryRaw<[{ count: bigint }]>`
      SELECT COUNT(*) as count
      FROM products p
      LEFT JOIN brands b ON p."brandId" = b.id
      WHERE
        p.status = 'ACTIVE'
        AND to_tsvector('english', p.name || ' ' || COALESCE(p."shortDescription", '') || ' ' || COALESCE(b.name, ''))
            @@ plainto_tsquery('english', ${query})
    `

    const enriched = results.map(r => ({
      ...r,
      image: imageMap.get(r.id) || null,
    }))

    return NextResponse.json({
      results: enriched,
      total: Number(countResult[0].count),
      page,
      totalPages: Math.ceil(Number(countResult[0].count) / limit),
    })
  } catch (err) {
    console.error('[SEARCH ERROR]', err)
    return NextResponse.json({ error: 'Search failed' }, { status: 500 })
  }
}
```

```typescript
  const recipients = [distributor.email, ...distributor.ccEmails].filter(Boolean)

  await resend.emails.send({
    from: process.env.EMAIL_FROM!,
    to: recipients,
    bcc: process.env.FULFILLMENT_BCC_EMAIL,
    subject: `New Order to Fulfill — Order #${order.orderNumber}`,
    react: FulfillmentNoticeEmail({ order, distributor, items }),
  })
}
```

## Sitemap and Robots

### `/app/sitemap.ts`

```typescript
import { MetadataRoute } from 'next'
import { prisma } from '@/lib/db/prisma'

const STORE_URL = process.env.NEXT_PUBLIC_STORE_URL!

export default async function sitemap(): Promise<MetadataRoute.Sitemap> {
  const [products, categories, collections, brands, posts, guides] = await Promise.all([
    prisma.product.findMany({ where: { status: 'ACTIVE' }, select: { slug: true, updatedAt: true } }),
    prisma.category.findMany({ where: { isActive: true }, select: { slug: true, updatedAt: true } }),
    prisma.collection.findMany({ where: { isActive: true }, select: { slug: true, updatedAt: true } }),
    prisma.brand.findMany({ where: { isActive: true }, select: { slug: true, updatedAt: true } }),
    prisma.blogPost.findMany({ where: { status: 'PUBLISHED' }, select: { slug: true, updatedAt: true } }),
    prisma.buyingGuide.findMany({ where: { status: 'PUBLISHED' }, select: { slug: true, updatedAt: true } }),
  ])

  const staticPages: MetadataRoute.Sitemap = [
    { url: STORE_URL, lastModified: new Date(), changeFrequency: 'daily', priority: 1 },
    { url: `${STORE_URL}/products`, lastModified: new Date(), changeFrequency: 'daily', priority: 0.9 },
    { url: `${STORE_URL}/collections`, lastModified: new Date(), changeFrequency: 'weekly', priority: 0.8 },
    { url: `${STORE_URL}/brands`, lastModified: new Date(), changeFrequency: 'weekly', priority: 0.7 },
    { url: `${STORE_URL}/guides`, lastModified: new Date(), changeFrequency: 'weekly', priority: 0.8 },
    { url: `${STORE_URL}/blog`, lastModified: new Date(), changeFrequency: 'daily', priority: 0.7 },
  ]

  return [
    ...staticPages,
    ...products.map(p => ({ url: `${STORE_URL}/products/${p.slug}`, lastModified: p.updatedAt, changeFrequency: 'weekly' as const, priority: 0.8 })),
    ...categories.map(c => ({ url: `${STORE_URL}/categories/${c.slug}`, lastModified: c.updatedAt, changeFrequency: 'weekly' as const, priority: 0.7 })),
    ...collections.map(c => ({ url: `${STORE_URL}/collections/${c.slug}`, lastModified: c.updatedAt, changeFrequency: 'weekly' as const, priority: 0.7 })),
    ...brands.map(b => ({ url: `${STORE_URL}/brands/${b.slug}`, lastModified: b.updatedAt, changeFrequency: 'monthly' as const, priority: 0.6 })),
    ...posts.map(p => ({ url: `${STORE_URL}/blog/${p.slug}`, lastModified: p.updatedAt, changeFrequency: 'monthly' as const, priority: 0.6 })),
    ...guides.map(g => ({ url: `${STORE_URL}/guides/${g.slug}`, lastModified: g.updatedAt, changeFrequency: 'monthly' as const, priority: 0.7 })),
  ]
}
```
